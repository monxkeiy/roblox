-- Functional Script
local currentCoord = "-"

GetBtn.MouseButton1Click:Connect(function()
    local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hrp then
        local pos = hrp.Position
        currentCoord = string.format("%.2f, %.2f, %.2f", pos.X, pos.Y, pos.Z)
        CoordLabel.Text = "Koordinat: " .. currentCoord
    else
        CoordLabel.Text = "Koordinat: Tidak Ditemukan"
    end
end)

CopyBtn.MouseButton1Click:Connect(function()
    if setclipboard and currentCoord ~= "-" then
        setclipboard(currentCoord)
        CoordLabel.Text = "Koordinat dicopy!"
    end
end)

-- Minimize Function
MinBtn.MouseButton1Click:Connect(function()
    Frame.Visible = false
    SettingsBtn.Visible = true
end)

SettingsBtn.MouseButton1Click:Connect(function()
    Frame.Visible = true
    SettingsBtn.Visible = false
end)
